 $(function(){
            $("#searchTxt").keyup(function(e){
                var searchTerm = $(this).val();
                if(!searchTerm){
                    $('td').show();
                }else{
                    searchTerm=searchTerm.trim().toLowerCase();
                    $('td').hide();
                    $('td').each(function(index,td){
                        //var rowNm=$(tr).data('rownm');
                          var rowNm = $(this).text();
                        //$(tr).attr('data-menuname') attr works for attr not for data
                        if(rowNm)
                        {
                            rowNm=rowNm.trim().toLowerCase();
                            if(rowNm.indexOf(searchTerm)>-1 || rowNm.indexOf(' '+searchTerm)>-1)
                            {
                                $(td).show();
                            }
                        }
                    });
                }
            });
        });
